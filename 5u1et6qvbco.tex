\begin{eqnarray}\label{eqn:net-likelihood}
\Delta L & =& L_+ - L_- \\
 & = \frac{r_+}{r_-} - \frac{r_-}{r_+} \\
& =  \frac{r+}{r_+}\frac{r_+}{r_-} - \frac{r_-}{r_-}\frac{r_-}{r_+} \\
& =  \frac{r_+^2 - r_-^2}{r_+r_-} \\
& =  \frac{(r_+ + r_-)(r_+ - r_-)}{r_+r_-} \\
& =  \frac{r_+ + r_-}{r_+r_-}\Delta r \\
\end{eqnarray}
