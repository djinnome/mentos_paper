{\mathcal P}_{+i}  & =  \frac{\frac{r_{+i}}{r_{-i}}}{\sum_j\left(\frac{r_{+j}}{r_{-j}} + \frac{r_{-j}}{r_{+j}}  \right)} & =  \frac{e^{\mathcal A_i/RT}}{\sum_j^n e^{\mathcal A_j/RT} + e^{-\mathcal A_j/RT}} \\
{\mathcal P}_{-i}  & = & \frac{\frac{r_{-i}}{r_{+i}}}{\sum_j\left(\frac{r_{+j}}{r_{-j}} + \frac{r_{-j}}{r_{+j}}  \right)} & = & \frac{e^{\mathcal -A_i/RT}}{\sum_j^n e^{\mathcal A_j/RT} + e^{-\mathcal A_j/RT}} \\
\end{eqnarray}
