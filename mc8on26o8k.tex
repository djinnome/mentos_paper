{\mathcal P}_{+i}  & =  \frac{r_{+i}/r_{-i}}{\sum_j^n\left(\frac{r_{+j}}{r_{-j}} + \frac{r_{-j}}{r_{+j}}  \right)} & =  \frac{e^{\mathcal A_i/RT}}{\sum_j^n \left( e^{\mathcal A_j/RT} + e^{-\mathcal A_j/RT}\right)} \\
{\mathcal P}_{-i}  & =  \frac{r_{-i}/r_{+i}}{\sum_j^n\left(\frac{r_{+j}}{r_{-j}} + \frac{r_{-j}}{r_{+j}}  \right)} & =  \frac{e^{\mathcal -A_i/RT}}{\sum_j^n \left(e^{\mathcal A_j/RT} + e^{-\mathcal A_j/RT}\right)} \\
\end{eqnarray}
